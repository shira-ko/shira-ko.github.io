          background(255);

          rect(ax, 0, width - ax, by);
          fill(0, 0, 255);
          rect(0, by, ax, height - by);
          fill(255, 255, 0);
          rect(cx, cy, width - cx, height - cy);
          stroke(0);
          strokeCap(SQUARE);
          strokeWeight(12);
          line(ax, 0, ax, height)
          line(0, by, width, by);
          line(cx, by, cx, height);
          strokeWeight(15);
          line(cx, cy, width, cy);
          strokeWeight(20);
          line(0, dy, ax, dy);
        }
      
        var origin = 195
        var canvasSq = 700
        function setup() {
          createCanvas(canvasSq, canvasSq);
        }

        function draw() {
          var x = width * 45 / origin;
        	var y = height * 145 / origin;
          var yx = width - (width * 18 / origin)
          var yy = y + (width-y) / 2
          var lx = x;
          var ly = width * 65 /origin;
          background(255);
          noStroke();
          fill(255, 0, 0);
          rect(x, 0, width - x, y);
          fill(0, 0, 255);
          rect(0, y, x, height);
          fill(255, 255, 0);
          rect(yx, yy, width - yx, height - yy);
          stroke(0);
          strokeCap(SQUARE);
          strokeWeight(8 * canvasSq / origin);
          line(x, 0, x, height);
          line(0, y, width, y);
          line(yx, y, yx, height);
          strokeWeight(10 * canvasSq / origin);
          line(yx, yy, width, yy);
          strokeWeight(14 * canvasSq / origin);
          line(0, ly, lx, ly);
        }
